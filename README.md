# bjectC_macro
巨集和標頭檔一樣，屬於預先編譯處理。

符號表示

         #define KeyWord string
         
有參數時的表示
 
         #define KeyWord(Param) string
         
實作範例

      #import <Foundation/Foundation.h>
      #define PI 3.14159
      
      int main()
      {
      
        @autoreleasepool {
        
           //...
        
        }     
      
        return 0;
        
      }
